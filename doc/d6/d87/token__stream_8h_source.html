213  bool consume(const T& expected, bool throw_on_error = false)
214  {

231  }
232 
233  m_pos++;

273  TokenStream<T> extract_until(const T& match, u32 end = END_OF_STREAM, bool level_aware = true, bool throw_on_error = false)
274  {
275  auto found = find_next(match, end, level_aware);
276  if (found > size() && throw_on_error)
277  {
278  throw TokenStreamException({"match Token '" + match + "' not found", get_current_line_number()});
279  }
280  auto end_pos = std::min(size(), found);
281  TokenStream res(m_increase_level_tokens, m_decrease_level_tokens);
282  res.m_data.reserve(res.m_data.size() + (end_pos - m_pos));
283  res.m_data.insert(res.m_data.begin(), m_data.begin() + m_pos, m_data.begin() + end_pos);
284  m_pos = end_pos;

303  Token<T> join_until(const T& match, const T& joiner, u32 end = END_OF_STREAM, bool level_aware = true, bool throw_on_error = false)
304  {
305  u32 start_line = get_current_line_number();
306  auto found = find_next(match, end, level_aware);
307  if (found > size() && throw_on_error)
308  {

411  u32 find_next(const T& match, u32 end = END_OF_STREAM, bool level_aware = true) const
412  {
413  u32 level = 0;
414  for (u32 i = m_pos; i < size() && i < end; ++i)
415  {
416  const auto& Token = at(i);
417  if ((!level_aware || level == 0) && Token == match)
418  {
419  return i;
420  }
421  else if (level_aware && std::find_if(m_increase_level_tokens.begin(), m_increase_level_tokens.end(), [&Token](const auto& x) { return Token == x; }) != m_increase_level_tokens.end())
422  {
423  level++;
424  }
425  else if (level_aware && level > 0
426  && std::find_if(m_decrease_level_tokens.begin(), m_decrease_level_tokens.end(), [&Token](const auto& x) { return Token == x; }) != m_decrease_level_tokens.end())
427  {
428  level--;
429  }
430  }
431  return end;
432  }
