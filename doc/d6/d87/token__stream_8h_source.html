229  {

248  m_pos++;

289  {

292  {

298  res.m_data.insert(res.m_data.begin(), m_data.begin() + m_pos, m_data.begin() + end_pos);

318  Token<T> join_until(const T& match, const T& joiner, u32 end = END_OF_STREAM, bool level_aware = true, bool throw_on_error = false)
319  {
320  u32 start_line = get_current_line_number();
321  auto found = find_next(match, end, level_aware);
322  if (found > size() && throw_on_error)
323  {

426  u32 find_next(const T& match, u32 end = END_OF_STREAM, bool level_aware = true) const
427  {
428  u32 level = 0;
429  for (u32 i = m_pos; i < size() && i < end; ++i)
430  {
431  const auto& Token = at(i);
432  if ((!level_aware || level == 0) && Token == match)
433  {
434  return i;
435  }
436  else if (level_aware && std::find_if(m_increase_level_tokens.begin(), m_increase_level_tokens.end(), [&Token](const auto& x) { return Token == x; }) != m_increase_level_tokens.end())
437  {
438  level++;
439  }
440  else if (level_aware && level > 0
441  && std::find_if(m_decrease_level_tokens.begin(), m_decrease_level_tokens.end(), [&Token](const auto& x) { return Token == x; }) != m_decrease_level_tokens.end())
442  {
443  level--;
444  }
445  }
446  return end;
447  }
