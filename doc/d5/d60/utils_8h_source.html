179  template<typename T>
180  CORE_API std::vector<T> split(const T& s, const char delim, bool obey_brackets = false)
181  {
182  std::vector<T> result;
183  T item = "";
184 
185  if (obey_brackets)
186  {
187  int bracket_level = 0;
188 
189  for (size_t i = 0; i < s.length(); ++i)
190  {
191  char c = s.at(i);
192  switch (c)
193  {
194  case '(':
195  case '{':
196  case '[':
197  ++bracket_level;
198  break;
199  case ')':
200  case '}':
201  case ']':
202  --bracket_level;
203  break;
204  default:
205  break;
206  }
207  if (bracket_level < 0)
208  {
209  bracket_level = 0;
210  }
211  if (c == delim)
212  {
213  // No constant expression, therefore not usable in switch case
214  if (bracket_level == 0)
215  {
216  result.push_back(item);
217  item = "";
218  }
219  else
220  {
221  item.push_back(c);
222  }
223  }
224  else
225  {
226  item.push_back(c);
227  }
228  }
229  if (!item.empty())
230  {
231  result.push_back(item);
232  }
233  }
234  else
235  {
236  std::stringstream ss(s);
237  while (std::getline(ss, item, delim))
238  {
239  result.push_back(item);
240  }
241  }
242  if (s.back() == delim)
243  {
244  result.push_back("");
245  }
246  return result;
247  }
