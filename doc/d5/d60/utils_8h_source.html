178  template<typename T>
179  CORE_API std::vector<T> split(const T& s, const char delim, bool obey_brackets = false)
180  {
181  std::vector<T> result;
182  T item = "";
183 
184  if (obey_brackets)
185  {
186  int bracket_level = 0;
187 
188  for (size_t i = 0; i < s.length(); ++i)
189  {
190  char c = s.at(i);
191  switch (c)
192  {
193  case '(':
194  case '{':
195  case '[':
196  ++bracket_level;
197  break;
198  case ')':
199  case '}':
200  case ']':
201  --bracket_level;
202  break;
203  default:
204  break;
205  }
206  if (bracket_level < 0)
207  {
208  bracket_level = 0;
209  }
210  if (c == delim)
211  {
212  // No constant expression, therefore not usable in switch case
213  if (bracket_level == 0)
214  {
215  result.push_back(item);
216  item = "";
217  }
218  else
219  {
220  item.push_back(c);
221  }
222  }
223  else
224  {
225  item.push_back(c);
226  }
227  }
228  if (!item.empty())
229  {
230  result.push_back(item);
231  }
232  }
233  else
234  {
235  std::stringstream ss(s);
236  while (std::getline(ss, item, delim))
237  {
238  result.push_back(item);
239  }
240  }
241  if (s.back() == delim)
242  {
243  result.push_back("");
244  }
245  return result;
246  }

325  {
326  auto s = str;

329  if (search.empty())

333
